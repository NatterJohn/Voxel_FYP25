private void SetHeights()
{
    heights = new float[width + 1, height + 1, width + 1];

    // Sphere parameters
    Vector3 center = new Vector3(width / 2f, height / 2f, width / 2f);
    float radius = Mathf.Min(width, height) / 2f;

    for (int x = 0; x < width + 1; x++)
    {
        for (int y = 0; y < height + 1; y++)
        {
            for (int z = 0; z < width + 1; z++)
            {
                Vector3 pos = new Vector3(x, y, z);
                float dist = Vector3.Distance(pos, center);

                // Base sphere field (normalized distance)
                float sphereField = dist / radius;

                // Add Perlin noise modulation
                float noise = PerlinNoise3D(
                    (float)x / width * noiseScale,
                    (float)y / height * noiseScale,
                    (float)z / width * noiseScale
                );

                // Combine sphere + noise
                heights[x, y, z] = sphereField + (noise * 0.2f); 
                // adjust multiplier for stronger/weaker bumps
            }
        }
    }
}
